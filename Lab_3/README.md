```SQL
-- initialize the stage to connect to s3
CREATE OR REPLACE STAGE TECHCATALYST_DE.EXTERNAL_STAGE.BEN_AWS_STAGE
        -- STORAGE_INTEGRATION = S3_INT
        URL='S3://BEN-TECHCATALYST-LABS/'
        CREDENTIALS = (AWS_KEY_ID= '', AWS_SECRET_KEY= '');

-- create a snowflake file format to parse csv files
CREATE OR REPLACE FILE FORMAT TECHCATALYST_DE.BCOMBS.CSV_FORMAT
TYPE = 'CSV'
FIELD_OPTIONALLY_ENCLOSED_BY = '"'
SKIP_HEADER = 1;

-- create a landing table for the data from s3
CREATE OR REPLACE TRANSIENT TABLE TECHCATALYST_DE.BCOMBS.TEST
(
    NAME STRING,
    PHONE_NUMBER STRING,
    CITY STRING,
    OCCUPATION STRING
);

-- PIPe CREATION
CREATE OR REPLACE PIPE TECHCATALYST_DE.BCOMBS.BEN_PIPE
AUTO_INGEST = TRUE
AS 
COPY INTO TECHCATALYST_DE.BCOMBS.TEST
FROM  @TECHCATALYST_DE.EXTERNAL_STAGE.BEN_AWS_STAGE
FILE_FORMAT = 'CSV_FORMAT';

-- show pipes
SHOW PIPES;
-- refresh the pip to pull new files
ALTER PIPE TECHCATALYST_DE.BCOMBS.BEN_PIPE REFRESH;

-- validate data pull
SELECT * FROM TECHCATALYST_DE.BCOMBS.TEST;
```
